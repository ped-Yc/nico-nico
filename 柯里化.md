        // 创建数组存储参数
        let _args = [...args];
        return foo;
      };

        return foo._args.reduce((acc, cur) => {
          return acc + cur;
        }, 0);
      };

      f1.getValue(); // 6

      var f3 = foo(1)(2)(3)(4);
